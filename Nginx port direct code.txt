Type into shell:

nano /etc/nginx/sites-available/<INSERT NAME OF APP>

and insert (change port to what you request and link to given LXC ip): 

server {
    listen 8006 ssl;
    ssl_certificate /root/proxmoxserver.tail380a97.ts.net.crt;
    ssl_certificate_key /root/proxmoxserver.tail380a97.ts.net.key;

    server_name proxmoxserver.tail380a97.ts.net;

    location / {
        proxy_pass https://192.168.100.2:8006;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
